\section*{Question 2}
Soient $q =
\begin{pmatrix}
  q_1\\
  q_2
\end{pmatrix}$,
$p =
\begin{pmatrix}
  p_1\\
  p_2
\end{pmatrix}$,
$f_1(q,p)$ et $f_2(q,p)$ tels que
\begin{align*}
  \dot{q} & = f_1(p)\\
  \dot{p} & = f_2(q).
\end{align*}

On calcule, en se rappelant des résultats de la question 1
\begin{align*}
  f_1(q,p) & =
  \begin{pmatrix}
    m_1\frac{p_1}{m_1}\\
    m_2\frac{p_2}{m_2}
  \end{pmatrix}\\
  & = p\\
%
  f_2(q,p) & = Gm_1m_2
  \begin{pmatrix}
    \frac{-q_1}{m_1\|q_1-q_2\|_2^3}\\
    \frac{q_2}{m_2\|q_1-q_2\|_2^3}
  \end{pmatrix}\\
  & = G
  \begin{pmatrix}
    \frac{-m_2q_1}{\|q_1-q_2\|_2^3}\\
    \frac{m_1q_2}{\|q_1-q_2\|_2^3}
  \end{pmatrix}.
\end{align*}
